@* Text input field using Component Library Styles as documented: Label: http://hmrc.github.io/assets-frontend/section-label.html Input http://hmrc.github.io/assets-frontend/section-text-input.html @param field Field The form field the input value represents @param args options arguments for the template '_labelText -> Text for the field label '_labelClass -> Label Modifier class '_inputClass -> CSS Class to add to input element '_currencyClass -> CSS Class to add to the currency element of the input '_modifierClass -> Modifier class for the input class, includes: input--xxsmall, input--xsmall, input--small, input--normal, input--fullwidth, input--no-spinner, input--cleared, input--link *@ @(field: Field, args: (Symbol, String)*)(implicit lang: Lang = Lang("en")) @import views.html.helper.FieldElements @import views.html.helpers.labelWrap @import views.html.helpers.forms.fieldErrors @argsMap = @{ args.toMap } @elements = @{ new FieldElements(field.id, field, null, argsMap, lang) } @value = @{ field.value match { case Some(x) => x case None => "" case x => x }} @wrapperClass = @{ var w = Seq.empty ++ argsMap.getOrElse('_labelClass, "").split(" ") w = if(argsMap.get('_modifierClass)) w ++ argsMap.get('_modifierClass).get.split(" ") else w w = if (field.hasErrors) w :+ "form-field--error" else w w.mkString(" ") } @inputClass = @{ var c:String = argsMap.getOrElse('_inputClass, "") c = if(field.hasErrors) c + " error-field" else c c.trim } @labelWrap('labelText -> argsMap.getOrElse('label, ""), 'for -> elements.field.name, '_after -> "true", '_modifierClass -> wrapperClass){ @fieldErrors(field) }
